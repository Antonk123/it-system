    if (e.key !== '/') return;
    const target = e.target as HTMLElement;
    const tag = target.tagName.toLowerCase();
    if (tag === 'input' || tag === 'textarea' || target.isContentEditable) return;

  return () => document.removeEventListener('keydown', handleKeyDown);
