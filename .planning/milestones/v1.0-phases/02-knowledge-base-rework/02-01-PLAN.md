```typescript
function stripHtml(html: string): string {
  return html
    .replace(/<[^>]*>/g, ' ')
    .replace(/&nbsp;/g, ' ')
    .replace(/&amp;/g, '&')
    .replace(/&lt;/g, '<')
    .replace(/&gt;/g, '>')
    .replace(/\s+/g, ' ')
    .trim();
}

const articles = db.prepare('SELECT rowid, title, content FROM kb_articles').all();
const stmt = db.prepare('INSERT INTO kb_articles_fts(rowid, title, content_plain) VALUES (?,?,?)');
const backfill = db.transaction(() => {
  for (const a of articles) {
    stmt.run(a.rowid, a.title, stripHtml(a.content));
  }
});
backfill();
```

**C. Replace GET /api/kb/articles search logic:**

When `search` query param is present AND non-empty after trim:
- Sanitize input: `const safeQuery = '"' + search.replace(/"/g, '""') + '"';`
- Use FTS5 MATCH query joining on rowid:
```sql
SELECT
  a.id, a.title, a.content, a.category_id, a.article_type, a.created_at, a.updated_at,
  c.name as category_name, c.color as category_color,
  snippet(kb_articles_fts, 1, '<mark>', '</mark>', '...', 25) AS snippet
FROM kb_articles_fts fts
JOIN kb_articles a ON a.rowid = fts.rowid
LEFT JOIN kb_categories c ON a.category_id = c.id
WHERE kb_articles_fts MATCH @search
  AND (@category_id IS NULL OR a.category_id = @category_id)
  AND (@article_type IS NULL OR a.article_type = @article_type)
ORDER BY rank
```
- Use named parameters: `{ search: safeQuery, category_id: category_id || null, article_type: article_type || null }`

When `search` is absent or empty:
- Use existing query but add `a.article_type` to SELECT and support `article_type` query param filter:
```sql
SELECT a.id, a.title, a.content, a.category_id, a.article_type, a.created_at, a.updated_at,
  c.name as category_name, c.color as category_color
FROM kb_articles a
LEFT JOIN kb_categories c ON a.category_id = c.id
WHERE 1=1
  AND (@category_id IS NULL OR a.category_id = @category_id)
  AND (@article_type IS NULL OR a.article_type = @article_type)
ORDER BY a.updated_at DESC
```

Read `article_type` from `req.query` alongside `search` and `category_id`.
