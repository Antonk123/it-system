**index.html — FOUC prevention script (per D-08):**

Add a blocking `<script>` inside `<head>`, just before the closing `</head>` tag. The script:

```html
<script>
  (function() {
    try {
      var mode = localStorage.getItem('app-mode-theme');
      document.documentElement.classList.add(
        (mode === 'light' || mode === 'dark') ? mode : 'dark'
      );
      var font = localStorage.getItem('app-font-theme');
      if (font && ['font-jakarta','font-crimson','font-libre','font-jetbrains'].indexOf(font) !== -1) {
        document.documentElement.classList.add(font);
      }
    } catch (e) {}
  })();
</script>
```

Place it after the last `<meta>` tag and before `</head>`. The font restoration is a 2-line addition per research recommendation — prevents font FOUC too with zero risk.

**src/hooks/useMode.ts — Reactive mode hook:**

Create a new file `src/hooks/useMode.ts`:

```typescript
import { useState, useEffect } from 'react';
import { getStoredMode, ModeTheme } from '@/lib/appearance';

/** Custom event name dispatched when mode changes in the same tab */
export const MODE_CHANGE_EVENT = 'app-mode-change';

/** Dispatch this from any code that calls applyMode + saveModeTheme */
export function dispatchModeChange(mode: ModeTheme) {
  window.dispatchEvent(new CustomEvent(MODE_CHANGE_EVENT, { detail: mode }));
}

/**
 * Reactive hook that returns the current ModeTheme.
 * Listens for same-tab custom events and cross-tab storage events.
 */
export function useMode(): ModeTheme {
  const [mode, setMode] = useState<ModeTheme>(getStoredMode);

  useEffect(() => {
    const handleCustom = (e: Event) => {
      const detail = (e as CustomEvent<ModeTheme>).detail;
      setMode(detail);
    };

    const handleStorage = (e: StorageEvent) => {
      if (e.key === 'app-mode-theme') {
        setMode((e.newValue as ModeTheme) ?? 'dark');
      }
    };

    window.addEventListener(MODE_CHANGE_EVENT, handleCustom);
    window.addEventListener('storage', handleStorage);

    return () => {
      window.removeEventListener(MODE_CHANGE_EVENT, handleCustom);
      window.removeEventListener('storage', handleStorage);
    };
  }, []);

  return mode;
}
```

This hook uses CustomEvent for same-tab reactivity (since StorageEvent only fires cross-tab). The `dispatchModeChange` helper will be called by the toggle in Plan 02.
